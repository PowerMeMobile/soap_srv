SendServiceSms

Sends service sms to recipient.

Test

The test form is only available for requests from the local machine.

Content-Length: length

<?xml version="1.0" encoding="utf-8"?>

  <soap:Body>
    <SendServiceSms xmlns="http://pmmsoapmessenger.com/">
      <customerID>int</customerID>
      <userName>string</userName>
      <userPassword>string</userPassword>
      <originator>string</originator>
      <serviceName>string</serviceName>
      <serviceUrl>string</serviceUrl>
      <recipientPhone>string</recipientPhone>
      <messageType>Latin or ArabicWithArabicNumbers or ArabicWithLatinNumbers</messageType>
      <defDate>string</defDate>
      <blink>boolean</blink>
      <flash>boolean</flash>
      <Private>boolean</Private>
    </SendServiceSms>
